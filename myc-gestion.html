// ==================== CONFIGURACIÓN ==================== const HOJA_CLIENTES = "Clientes"; const HOJA_MOVIMIENTOS = "Movimientos"; const HOJA_VENTAS = "Ventas"; // Nueva hoja para ventas detalladas const HOJA_GASTOS = "Gastos"; // Nueva hoja para gastos const HOJA_FABRICAS = "Fabricas"; // Nueva hoja para cargas/devoluciones function doGet(e) { const ruta = e.parameter.ruta || ''; if (ruta === 'clientes') return ContentService.createTextOutput(JSON.stringify(obtenerClientes())).setMimeType(ContentService.MimeType.JSON); if (ruta === 'movimientos') return ContentService.createTextOutput(JSON.stringify(obtenerMovimientos())).setMimeType(ContentService.MimeType.JSON); if (ruta === 'ventas') return ContentService.createTextOutput(JSON.stringify(obtenerVentas())).setMimeType(ContentService.MimeType.JSON); if (ruta === 'gastos') return ContentService.createTextOutput(JSON.stringify(obtenerGastos())).setMimeType(ContentService.MimeType.JSON); if (ruta === 'fabricas') return ContentService.createTextOutput(JSON.stringify(obtenerFabricas())).setMimeType(ContentService.MimeType.JSON); return ContentService.createTextOutput(JSON.stringify({ error: 'Ruta no válida' })).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const data = e.parameter || {}; const ruta = data.ruta || ''; if (ruta === 'agregarCliente') return agregarCliente(data); if (ruta === 'eliminarCliente') return eliminarCliente(data); if (ruta === 'agregarMovimiento') return agregarMovimiento(data); if (ruta === 'eliminarMovimiento') return eliminarMovimiento(data); if (ruta === 'agregarVenta') return agregarVenta(data); if (ruta === 'eliminarVenta') return eliminarVenta(data); if (ruta === 'agregarGasto') return agregarGasto(data); if (ruta === 'eliminarGasto') return eliminarGasto(data); if (ruta === 'agregarFabrica') return agregarFabrica(data); if (ruta === 'eliminarFabrica') return eliminarFabrica(data); return ContentService.createTextOutput(JSON.stringify({ error: 'Ruta no válida' })).setMimeType(ContentService.MimeType.JSON); } // ==================== FUNCIONES ==================== function obtenerClientes() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_CLIENTES); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; return data.slice(1).map(row => row[0]).filter(Boolean); } function obtenerMovimientos() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_MOVIMIENTOS); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; return data.slice(1).map(row => ({ id: row[0], fecha: row[1], cliente_nombre: row[2], tipo: row[3], monto: row[4], forma_pago: row[5], detalle: row[6] })); } function obtenerVentas() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_VENTAS); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; return data.slice(1).map(row => ({ id: row[0], fecha: row[1], cliente: row[2], c17: row[3] || 0, c35: row[4] || 0, c14: row[5] || 0, c20: row[6] || 0, c32: row[7] || 0, total: row[8] || 0, pago: row[9], detalle: row[10] })); } function obtenerGastos() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_GASTOS); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; return data.slice(1).map(row => ({ id: row[0], fecha: row[1], categoria: row[2], monto: row[3], nota: row[4] })); } function obtenerFabricas() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_FABRICAS); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; return data.slice(1).map(row => ({ id: row[0], fecha: row[1], fabrica: row[2], tipo: row[3], cargada: row[4] || 0, devuelta: row[5] || 0 })); } // ==================== CRUD ==================== function agregarCliente(data) { const nombre = data.nombre; if (!nombre) return { error: 'Nombre requerido' }; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_CLIENTES); if (!sheet) return { error: 'Hoja no encontrada' }; sheet.appendRow([nombre]); return { success: true }; } function eliminarCliente(data) { const nombre = data.nombre; if (!nombre) return { error: 'Nombre requerido' }; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_CLIENTES); if (!sheet) return { error: 'Hoja no encontrada' }; const dataRange = sheet.getDataRange().getValues(); for (let i = dataRange.length - 1; i >= 1; i--) { if (dataRange[i][0] === nombre) { sheet.deleteRow(i + 1); return { success: true }; } } return { error: 'Cliente no encontrado' }; } function agregarMovimiento(data) { const { fecha, cliente, tipo, monto, formaPago, detalle } = data; if (!fecha || !cliente || !tipo || !monto) return { error: 'Datos incompletos' }; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_MOVIMIENTOS); if (!sheet) return { error: 'Hoja no encontrada' }; const id = 'm' + Date.now(); sheet.appendRow([id, fecha, cliente, tipo, monto, formaPago || '', detalle || '']); return { success: true, id }; } function eliminarMovimiento(data) { const id = data.id; if (!id) return { error: 'ID requerido' }; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_MOVIMIENTOS); if (!sheet) return { error: 'Hoja no encontrada' }; const dataRange = sheet.getDataRange().getValues(); for (let i = dataRange.length - 1; i >= 1; i--) { if (dataRange[i][0] === id) { sheet.deleteRow(i + 1); return { success: true }; } } return { error: 'Movimiento no encontrado' }; } // FUNCIONES SIMILARES PARA ventas, gastos y fabricas... // (Se pueden implementar siguiendo el mismo patrón)

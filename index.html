function imprimirSeccion(sector) { // Crear un contenedor temporal para la sección a imprimir let contenedor = document.createElement('div'); contenedor.id = 'seccion-imprimir'; let titulo = ''; if (sector === 'resumen') { const card = document.querySelector('#tab-resumen .card').cloneNode(true); // Filtrar clientes con saldo positivo const items = card.querySelectorAll('.resumen-item'); let totalDeuda = 0; items.forEach(item => { const spanSaldo = item.querySelector('.saldo-pos, .saldo-neg, .saldo-cero'); if (!spanSaldo || !spanSaldo.classList.contains('saldo-pos')) { item.remove(); } else { const montoTexto = spanSaldo.innerText.match(/\$([\d.,]+)/); if (montoTexto) { totalDeuda += parseFloat(montoTexto[1].replace(/\./g,'').replace(',','.')); } } }); const totalBox = card.querySelector('.total-box'); if (totalBox) { totalBox.innerHTML = `💰 Total deudas: $${totalDeuda.toLocaleString('es-AR')}`; } if (card.querySelectorAll('.resumen-item').length === 0) { const lista = card.querySelector('#resumenLista'); if (lista) lista.innerHTML = '
No hay clientes con deuda activa
'; } // Quitar botones de impresión const btns = card.querySelectorAll('.btn-sm'); btns.forEach(btn => btn.remove()); contenedor.appendChild(card); titulo = 'Resumen de Deudas'; } else if (sector === 'movimientos') { const card = document.querySelector('#tab-movimientos .card').cloneNode(true); const filtros = card.querySelector('.filtros'); if (filtros) filtros.remove(); const btns = card.querySelectorAll('.btn-sm'); btns.forEach(btn => btn.remove()); contenedor.appendChild(card); titulo = 'Historial de Movimientos'; } else if (sector === 'clientes') { const card = document.querySelector('#tab-clientes .card:last-child').cloneNode(true); const btns = card.querySelectorAll('.btn-sm'); btns.forEach(btn => btn.remove()); contenedor.appendChild(card); titulo = 'Lista de Clientes'; } // Limpiar cualquier sección anterior y agregar la nueva const existing = document.getElementById('seccion-imprimir'); if (existing) existing.remove(); document.body.appendChild(contenedor); // Aplicar modo impresión document.body.classList.add('printing'); // Ajustar estilos para que se vea bien en impresión const style = document.createElement('style'); style.textContent = ` body.printing #seccion-imprimir { display: block; margin: 20px; background: white; } body.printing #seccion-imprimir .card { box-shadow: none; border: 1px solid #ccc; page-break-inside: avoid; } body.printing .total-box { -webkit-print-color-adjust: exact; print-color-adjust: exact; } body.printing table { width: 100%; border-collapse: collapse; } body.printing th, body.printing td { border: 1px solid #ccc; padding: 8px; } `; document.head.appendChild(style); // Llamar a imprimir window.print(); // Restaurar setTimeout(() => { document.body.classList.remove('printing'); style.remove(); contenedor.remove(); }, 100); }
